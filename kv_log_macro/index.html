Log macro for log’s kv-unstable backend.

Example

use kv_log_macro::info;

femme::start(log::LevelFilter::Info).unwrap();

info!("hello");
info!("hello",);
info!("hello {}", "cats");
info!("hello {}", "cats",);
info!("hello {}", "cats", {
    cat_1: "chashu",
    cat_2: "nori",
});
